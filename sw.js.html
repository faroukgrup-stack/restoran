// 1. تعريف اسم الكاش (قم بتغيير v1 إلى v2 عند تحديث الموقع مستقبلاً) const CACHE_NAME = 'fg-restaurant-v2'; // 2. قائمة الملفات التي سيتم حفظها للعمل بدون إنترنت const ASSETS = [ '/', '/index.html', // أضف هنا أي ملفات CSS أو أيقونات إذا كانت منفصلة ]; // مرحلة التثبيت: حفظ الملفات في ذاكرة المتصفح self.addEventListener('install', (e) => { e.waitUntil( caches.open(CACHE_NAME).then((cache) => { console.log('جاري تثبيت ملفات الكاش الجديدة...'); return cache.addAll(ASSETS); }).then(() => self.skipWaiting()) // تفعيل التحديث فوراً ); }); // مرحلة التنشيط: حذف النسخ القديمة من الكاش تماماً self.addEventListener('activate', (e) => { e.waitUntil( caches.keys().then((cacheNames) => { return Promise.all( cacheNames.map((cache) => { if (cache !== CACHE_NAME) { console.log('جاري حذف النسخة القديمة من الكاش:', cache); return caches.delete(cache); } }) ); }).then(() => self.clients.claim()) // السيطرة على المتصفح فوراً بالنسخة الجديدة ); }); // استراتيجية الجلب (الشبكة أولاً): يضمن الحصول على أحدث المعادلات والأرقام // وإذا كان المستخدم أوفلاين، يفتح الموقع من الكاش self.addEventListener('fetch', (e) => { e.respondWith( fetch(e.request).catch(() => { return caches.match(e.request); }) ); });
